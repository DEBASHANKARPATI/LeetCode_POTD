class Solution {
public:
    int MaxOr = INT_MIN;
    int Count = 0;
    void Solve(vector<int>& nums,int index,int BitOr)
    {
      if(index<0) {
         if(MaxOr<BitOr)
         {
            MaxOr = BitOr;
            Count=1;
         }
         else if(MaxOr==BitOr)
         {
            Count++;
         }
        return;
      }
      Solve(nums,index-1,BitOr|nums[index]);
      Solve(nums,index-1,BitOr);
    }
    int countMaxOrSubsets(vector<int>& nums) {
        int n = nums.size()-1;
        Solve(nums,n,0);
        return Count; 
    }
};
